{{- /* Render an example chart file verbatim (read from the real file via a module mount) so the docs can never drift from a runnable chart. Highlighting is chosen from the path: templates/*.yaml and *.tpl are Helm/Go templates (highlight the template logic), everything else by extension. Usage: {{< chartfile "hero/templates/deployment.yaml" >}} {{< chartfile path="hero/values.yaml" lang="yaml" title="values.yaml" >}} */ -}} {{- $path := .Get "path" | default (.Get 0) -}} {{- $res := resources.Get (printf "examples/%s" $path) -}} {{- if not $res -}} {{- errorf "chartfile: examples/%s not found" $path -}} {{- end -}} {{- $base := path.Base $path -}} {{- $ext := strings.TrimPrefix "." (path.Ext $path) -}} {{- $isTemplate := or (strings.HasSuffix $path ".tpl") (and (in $path "/templates/") (or (eq $ext "yaml") (eq $ext "yml"))) -}} {{- $lang := .Get "lang" -}} {{- if not $lang -}} {{- if $isTemplate -}}{{- $lang = "gotmpl" -}} {{- else if or (eq $ext "yaml") (eq $ext "yml") -}}{{- $lang = "yaml" -}} {{- else if eq $ext "json" -}}{{- $lang = "json" -}} {{- else if eq $ext "toml" -}}{{- $lang = "toml" -}} {{- else -}}{{- $lang = "text" -}}{{- end -}} {{- end -}} {{- $title := .Get "title" | default $base -}}
{{ $title }}
{{ transform.Highlight (strings.TrimSpace $res.Content) $lang }}
